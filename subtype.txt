(define bar
  (fun ([:x (Int (< x 5))] [-> Int])
    x))
    
 (define call-bar
  (fun ([:y Int] [-> Int])
    (cond 
      ([< y 4] (bar 4)) --bar可以从(< y 4), 这里推断出(bar 4)是可以安全调用的
      [else (bar 5)])))

  (fun ([:x B] [-> A])
    (x.a)))
